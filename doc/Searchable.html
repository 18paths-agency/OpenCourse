def self._find_all_by table:, term:
  begin
    table
      .titleize
      .constantize
      .where("title like '%#{term}%'")
  rescue
    []
  end

def self.find_all_by search_term
  related_tables.map do |table|
    self._find_all_by table: table, term: search_term
  end.compact.flatten
